Setup Git
$ git config --global user.name "<name>"
$ git config --global user.email <email>

Initialize git repo
$ git init

Project status
$ git status

Add file to stage
$ git add <filename>
	options:
	*.txt //txt files
	* //all files

Commit file
$ git commit -m "message"
	options:
	-m "message" //add message to commit
	-a //add files to stage

Check repo history
$ git log
	options:
	--graph //brach graph
	--oneline
	-p //changes
	-1 //number of commits
	--decorate

Create alias
$ git config --global alias.lg 'log --graph --oneline --decorate --all'

Differences
$ git diff
	options
	--staged //changes between stage and last commit

Reset local changes
$ git checkout -- <filename> //reset local changes on file

Rset the Stage
$ git reset HEAD
	options
	<filename> //unstage file

Remove commit
$ git reset <commit>
	options
	--hard // remove changes

Amend last commit
$ git commit --amend

Create branch
$ git branch 

Change to branch
$ git checkout <branchname>

Check current branch
$ git branch
